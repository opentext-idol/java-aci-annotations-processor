15  /**
16   * Indicates that when the given IDOL field is reached a {@code StAXProcessor} with the given class should be called.
17   * <p>
18   * The method should take a single argument.
19   * <p>
20   * If no class is specified then the processor that is used will be generated via
21   * {@link com.autonomy.aci.client.annotations.IdolAnnotationsProcessorFactory#forClass}
22   * for the type of the first argument of the method. In this case the type of the first argument of the method should be
23   * annotated with IDOL processor annotations.
24   * <p>
25   * Note: if your class depends upon annotated methods in a non-public superclass, the annotations will not be found. This
26   * is a limitation of the Java Reflection libraries.
27   */
28  @Retention(RetentionPolicy.RUNTIME)
29  @Target(ElementType.METHOD)
30  public @interface IdolProcessorField {
31  
32      /**
33       * @return The root field of the annotated class. If {@link com.autonomy.aci.client.annotations.IdolProcessorField#fieldType fieldType}
34       * is {@link com.autonomy.aci.client.annotations.FieldType#PATH PATH} then
35       * the path is relative to the root IDOL field for the class.
36       */
37      String value();
38  
39      /**
40       * @return The class of the {@code StAXProcessor} to be used. If no value is given then a processor will be generated
41       * using {@link com.autonomy.aci.client.annotations.IdolAnnotationsProcessorFactory#forClass}
42       */
43      Class<? extends StAXProcessor<?>> processor() default ProcessorFieldDefaultProcessor.class;
44  
45      /**
46       * @return The {@link com.autonomy.aci.client.annotations.ValueType} of the
47       * {@link com.autonomy.aci.client.annotations.IdolProcessorField#value value}
48       */
49      ValueType valueType() default ValueType.LITERAL;
50  
51      /**
52       * @return The {@link com.autonomy.aci.client.annotations.FieldType} of the
53       * {@link com.autonomy.aci.client.annotations.IdolProcessorField#value value}
54       */
55      FieldType fieldType() default FieldType.FIELD;
56  
57      /**
58       * @return The defaultValue of the value if {@link com.autonomy.aci.client.annotations.IdolProcessorField#valueType valueType} is
59       * {@link com.autonomy.aci.client.annotations.ValueType#CONFIGURED CONFIGURED} and no
60       * entry for the key given in value is found
61       */
62      String defaultValue() default "";

64  }
