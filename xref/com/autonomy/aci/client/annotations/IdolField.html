6   package com.autonomy.aci.client.annotations;

8   import java.lang.annotation.ElementType;
9   import java.lang.annotation.Retention;
10  import java.lang.annotation.RetentionPolicy;
11  import java.lang.annotation.Target;
12  
13  /**
14   * Indicates that a method should be called with the result of an IDOL field. The method should take a single argument.
15   * <p>
16   * If the first argument has type {@code String} the method will be called with the value of the IDOL field.
17   * <p>
18   * If the first argument is a primitive type or boxed wrapper class the value of the IDOL field will be converted to
19   * the correct type.
20   * <ul>
21   * <li> For {@code char}/{@code Character}, an {@code IllegalArgumentException} will be thrown if the field value
22   * does not have length 1.
23   * <li> For numeric types, if the field value cannot be converted then 0 will be returned.  This includes overflow cases.
24   * It is the client's responsibility to ensure the chosen numeric type is large enough to handle all possible values.
25   * </ul>
26   * <p>
27   * If the first argument as a constructor that takes a single string then that constructor will be called with the value
28   * of the IDOL field.
29   * <p>
30   * Note: if your class depends upon annotated methods in a non-public superclass, the annotations will not be found.  This
31   * is a limitation of the Java Reflection libraries
32   */
33  @Retention(RetentionPolicy.RUNTIME)
34  @Target(ElementType.METHOD)
35  public @interface IdolField {
36  
37      /**
38       * @return The root field of the annotated class. If {@link com.autonomy.aci.client.annotations.IdolField#fieldType fieldType}
39       * is {@link com.autonomy.aci.client.annotations.FieldType#PATH PATH} then the path is relative to the root IDOL
40       * field for the class.
41       */
42      String value();
43  
44      /**
45       * If present, the value will be read from the attribute of the IDOL field with the given name, instead of the IDOL
46       * field value.
47       * @return The attribute name to read the value from
48       */
49      String attributeName() default "";
50  
51      /**
52       * @return The {@link com.autonomy.aci.client.annotations.ValueType} of the
53       * {@link com.autonomy.aci.client.annotations.IdolField#value value}
54       */
55      ValueType valueType() default ValueType.LITERAL;
56  
57      /**
58       * @return The {@link com.autonomy.aci.client.annotations.FieldType} of the
59       * {@link com.autonomy.aci.client.annotations.IdolField#value value}
60       */
61      FieldType fieldType() default FieldType.FIELD;
62  
63      /**
64       * @return The defaultValue of the value if {@link com.autonomy.aci.client.annotations.IdolField#valueType valueType}
65       * is {@link com.autonomy.aci.client.annotations.ValueType#CONFIGURED CONFIGURED} and no entry for the key given in
66       * value is found
67       */
68      String defaultValue() default "";
69  
70  }
